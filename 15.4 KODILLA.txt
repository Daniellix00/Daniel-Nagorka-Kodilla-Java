SELECT ISSUES.*, ISSUELIST.NAME AS LIST_NAME
FROM ISSUES, ISSUELIST
WHERE ISSUES.ISSUESLIST_ID = ISSUELIST.ID;
SELECT ISSUES.*, U.FIRSTNAME, U.LASTNAME
FROM ISSUES
JOIN USERS U on ISSUES.USER_ID_ASSIGNEDTO = U.ID;
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS USERS_TASKS
FROM USERS U JOIN ISSUES I ON U.ID = I.ISSUESLIST_ID
GROUP BY U.FIRSTNAME, U.LASTNAME
